<php

namespace DayleRees\ColourSchemes;

class Colour
{
    /**
     * Theme name.

     * @type string
     */
    protected $theme = 'spearmint';

    /**
     * Dayle Rees Colour Schemes

     * @return ColourScheme
     */
    public function schemes()
    {
        $theme = new ColourScheme;
        $theme->setAwesome('100%');
        return $theme;
    }
}
